# kernel.ea
kernel scale(src: *f32, dst: *mut f32, factor: f32)[256] {
    let v = vload(src)
    vstore(dst, v .* vbroadcast(factor))
}

# Python
import ea, numpy as np

kernel = ea.load("kernel.ea")
src = np.random.randn(1_000_000).astype(np.float32)
dst = np.empty_like(src)
kernel.scale(src, dst, factor=2.0)

ea.load("kernel.ea") compiles your kernel to native AVX2 or NEON, caches it, and returns a callable Python function. Not auto-vectorized. You write the SIMD, you get the SIMD. Works on Linux x86_64, Linux aarch64, and Windows x86_64.
