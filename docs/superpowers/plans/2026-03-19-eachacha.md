# eachacha Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone ChaCha20 crypto demo (RFC 7539) in Ea SIMD kernels with fused encrypt+stats, benchmarked honestly against NumPy, generic C, and OpenSSL.

**Architecture:** New repo at `/root/dev/eachacha`. Two Ea kernels (chacha20 + fused), one C reference, Python test suite + benchmark. ChaCha20 state stored as 4 x i32x4 row vectors; column rounds operate directly, diagonal rounds use shuffle to rotate rows. Byte-level XOR uses dual-pointer trick: caller passes the same 64-byte scratch buffer as both `*mut i32` and `*u8`, since Ea has no pointer casting.

**Tech Stack:** Ea compiler (ea binary from eacompute), Python 3, NumPy, `cryptography` (OpenSSL wrapper), cc (C compiler)

**Spec:** `docs/superpowers/specs/2026-03-19-eachacha-design.md`

**Ea language constraints (verified):**
- Vector literals: `[1, 2, 3, 4]i32x4` (bracket + type suffix)
- No pointer casting between types (`*u8` cannot become `*i32`)
- No scalar bitwise operators (`^`, `&`, `|`) — only vector `.^`, `.&`, `.|`
- `store(ptr, offset, value)` / `load(ptr, offset)` — offset is in ELEMENTS, not bytes
- `reduce_min` / `reduce_max` work on any vector type including u8x16
- `if` is a statement, not an expression (no `let x = if ...`)
- `load_masked(ptr, offset, count)` / `store_masked(ptr, offset, value, count)` for partial vectors
- `func` (no `export`) creates internal functions callable within the same file

### Task 1: ChaCha20 kernel — quarter round and encryption

The core SIMD kernel. State is 4 x i32x4 row vectors. The `chacha20_encrypt` function takes a 64-byte scratch buffer as both `*mut i32` (for writing keystream) and `*u8` (for reading keystream as bytes to XOR with plaintext). The caller passes the same allocation through both parameters.

```ea
// chacha20.ea — RFC 7539 ChaCha20 SIMD kernel
//
// State: 4 x i32x4 rows. Column rounds operate directly.
// Diagonal rounds: shuffle rows to align, QR, shuffle back.
//
// Encryption uses dual-pointer trick for keystream buffer:
//   ks_i32 (*mut i32) and ks_u8 (*u8) point to the same 64-byte allocation.
//   Ea has no pointer casting, so the caller provides both views.

// --- Rotation helpers (internal) ---

func rotl16(v: i32x4) -> i32x4 {
    return (v .<< splat(16)) .| (v .>> splat(16))
}

func rotl12(v: i32x4) -> i32x4 {
    return (v .<< splat(12)) .| (v .>> splat(20))
}

func rotl8(v: i32x4) -> i32x4 {
    return (v .<< splat(8)) .| (v .>> splat(24))
}

func rotl7(v: i32x4) -> i32x4 {
    return (v .<< splat(7)) .| (v .>> splat(25))
}

// --- Double round: column QR + diagonal QR ---

func double_round(
    r0: *mut i32, r1: *mut i32, r2: *mut i32, r3: *mut i32
) {
    let mut a: i32x4 = load(r0, 0)
    let mut b: i32x4 = load(r1, 0)
    let mut c: i32x4 = load(r2, 0)
    let mut d: i32x4 = load(r3, 0)

    // Column round
    a = a .+ b
    d = rotl16(d .^ a)
    c = c .+ d
    b = rotl12(b .^ c)
    a = a .+ b
    d = rotl8(d .^ a)
    c = c .+ d
    b = rotl7(b .^ c)

    // Prepare for diagonal round
    b = shuffle(b, [1, 2, 3, 0])
    c = shuffle(c, [2, 3, 0, 1])
    d = shuffle(d, [3, 0, 1, 2])

    // Diagonal round
    a = a .+ b
    d = rotl16(d .^ a)
    c = c .+ d
    b = rotl12(b .^ c)
    a = a .+ b
    d = rotl8(d .^ a)
    c = c .+ d
    b = rotl7(b .^ c)

    // Undo diagonal rotation
    b = shuffle(b, [3, 0, 1, 2])
    c = shuffle(c, [2, 3, 0, 1])
    d = shuffle(d, [1, 2, 3, 0])

    store(r0, 0, a)
    store(r1, 0, b)
    store(r2, 0, c)
    store(r3, 0, d)
}

// ChaCha20 block function. Exported for test vector verification.
// Writes 16 x i32 keystream words to out.
export func chacha20_block(
    key: *i32, nonce: *i32, counter: i32,
    out: *mut i32
) {
    // Init state
    let c: i32x4 = [0x61707865, 0x3320646e, 0x79622d32, 0x6b206574]i32x4
    let k0: i32x4 = load(key, 0)
    let k1: i32x4 = load(key, 4)
    let cn: i32x4 = [counter, nonce[0], nonce[1], nonce[2]]i32x4

    // Working state in out buffer (used as scratch)
    store(out, 0, c)
    store(out, 4, k0)
    store(out, 8, k1)
    store(out, 12, cn)

    // 10 double rounds
    let mut i: i32 = 0
    while i < 10 {
        // Pass pointers to each row (4 i32 elements apart)
        // Note: double_round reads/writes via load/store at offset 0
        // We need row pointers. Since Ea can't do ptr arithmetic,
        // we inline the rounds here instead.
        let mut row0: i32x4 = load(out, 0)
        let mut row1: i32x4 = load(out, 4)
        let mut row2: i32x4 = load(out, 8)
        let mut row3: i32x4 = load(out, 12)

        // Column round
        row0 = row0 .+ row1
        row3 = rotl16(row3 .^ row0)
        row2 = row2 .+ row3
        row1 = rotl12(row1 .^ row2)
        row0 = row0 .+ row1
        row3 = rotl8(row3 .^ row0)
        row2 = row2 .+ row3
        row1 = rotl7(row1 .^ row2)

        // Diagonal prep
        row1 = shuffle(row1, [1, 2, 3, 0])
        row2 = shuffle(row2, [2, 3, 0, 1])
        row3 = shuffle(row3, [3, 0, 1, 2])

        // Diagonal round
        row0 = row0 .+ row1
        row3 = rotl16(row3 .^ row0)
        row2 = row2 .+ row3
        row1 = rotl12(row1 .^ row2)
        row0 = row0 .+ row1
        row3 = rotl8(row3 .^ row0)
        row2 = row2 .+ row3
        row1 = rotl7(row1 .^ row2)

        // Undo diagonal
        row1 = shuffle(row1, [3, 0, 1, 2])
        row2 = shuffle(row2, [2, 3, 0, 1])
        row3 = shuffle(row3, [1, 2, 3, 0])

        store(out, 0, row0)
        store(out, 4, row1)
        store(out, 8, row2)
        store(out, 12, row3)

        i = i + 1
    }

    // Add initial state
    store(out, 0, load(out, 0) .+ c)
    store(out, 4, load(out, 4) .+ k0)
    store(out, 8, load(out, 8) .+ k1)
    store(out, 12, load(out, 12) .+ cn)
}

// ChaCha20 counter-mode encryption.
// ks_i32 and ks_u8 MUST point to the same 64-byte buffer.
// ks_i32 is used to write keystream as i32x4, ks_u8 to read it as u8x16.
export func chacha20_encrypt(
    key: *i32, nonce: *i32, counter: i32,
    plaintext: *u8, ciphertext: *mut u8, len: i32,
    ks_i32: *mut i32, ks_u8: *u8
) {
    let mut offset: i32 = 0
    let mut ctr: i32 = counter

    // Full 64-byte blocks
    while offset + 64 <= len {
        // Generate keystream block into ks_i32
        chacha20_block(key, nonce, ctr, ks_i32)

        // XOR plaintext with keystream (read keystream as u8x16 via ks_u8)
        let pt0: u8x16 = load(plaintext, offset)
        let ks0: u8x16 = load(ks_u8, 0)
        store(ciphertext, offset, pt0 .^ ks0)

        let pt1: u8x16 = load(plaintext, offset + 16)
        let ks1: u8x16 = load(ks_u8, 16)
        store(ciphertext, offset + 16, pt1 .^ ks1)

        let pt2: u8x16 = load(plaintext, offset + 32)
        let ks2: u8x16 = load(ks_u8, 32)
        store(ciphertext, offset + 32, pt2 .^ ks2)

        let pt3: u8x16 = load(plaintext, offset + 48)
        let ks3: u8x16 = load(ks_u8, 48)
        store(ciphertext, offset + 48, pt3 .^ ks3)

        ctr = ctr + 1
        offset = offset + 64
    }

    // Partial final block (< 64 bytes remaining)
    if offset < len {
        chacha20_block(key, nonce, ctr, ks_i32)
        let remaining: i32 = len - offset

        // Full u8x16 chunks within the tail
        let mut tail_off: i32 = 0
        while tail_off + 16 <= remaining {
            let pt: u8x16 = load(plaintext, offset + tail_off)
            let ks: u8x16 = load(ks_u8, tail_off)
            store(ciphertext, offset + tail_off, pt .^ ks)
            tail_off = tail_off + 16
        }

        // Final masked chunk (< 16 bytes)
        if tail_off < remaining {
            let left: i32 = remaining - tail_off
            let pt: u8x16 = load_masked(plaintext, offset + tail_off, left)
            let ks: u8x16 = load_masked(ks_u8, tail_off, left)
            store_masked(ciphertext, offset + tail_off, pt .^ ks, left)
        }
    }
}
```

**Key design decisions:**
- `chacha20_block` is exported for direct test vector verification
- `chacha20_encrypt` takes `ks_i32`/`ks_u8` dual-pointer to same scratch buffer (no pointer casting needed)
- Tail uses `load_masked`/`store_masked` — full SIMD, no scalar XOR needed
- All rotations are inline helper functions (no `export`)
- Double round is inlined in the loop (Ea can't do pointer arithmetic for row pointers)

```python
import ctypes as ct
import numpy as np

from pathlib import Path

_lib = ct.CDLL(str(Path(__file__).parent / "libchacha20.so"))

    _lib.chacha20_block.argtypes = [

def to_i32_array(values):
    arr = (ct.c_int32 * len(values))()
    for i, v in enumerate(values):
        arr[i] = ct.c_int32(v & 0xFFFFFFFF).value
    return arr

def make_scratch():
    """Create 64-byte scratch buffer with both i32 and u8 views."""
    buf = (ct.c_uint8 * 64)()
    i32_ptr = ct.cast(buf, ct.POINTER(ct.c_int32))
    u8_ptr = ct.cast(buf, ct.POINTER(ct.c_uint8))
    return i32_ptr, u8_ptr

    """Cross-verify with OpenSSL."""
    print("Test: Cross-verification with OpenSSL...", end=" ")
    try:
        from cryptography.hazmat.primitives.ciphers import Cipher, algorithms
    except ImportError:
        print("SKIP (cryptography not installed)")

        return False

    print("PASS")
    return True
```

```c
/* chacha20_ref.c — RFC 7539 ChaCha20, plain C, no SIMD. cc -O3 */
#include <stdint.h>
#include <string.h>

#define ROTL32(v, n) (((v) << (n)) | ((v) >> (32 - (n))))

#define QR(a, b, c, d) \
    a += b; d ^= a; d = ROTL32(d, 16); \
    c += d; b ^= c; b = ROTL32(b, 12); \
    a += b; d ^= a; d = ROTL32(d, 8);  \
    c += d; b ^= c; b = ROTL32(b, 7);

static void chacha20_block(uint32_t out[16], const uint32_t key[8],
                           uint32_t counter, const uint32_t nonce[3]) {
    uint32_t s[16];
    s[0]=0x61707865; s[1]=0x3320646e; s[2]=0x79622d32; s[3]=0x6b206574;
    s[4]=key[0]; s[5]=key[1]; s[6]=key[2]; s[7]=key[3];
    s[8]=key[4]; s[9]=key[5]; s[10]=key[6]; s[11]=key[7];
    s[12]=counter; s[13]=nonce[0]; s[14]=nonce[1]; s[15]=nonce[2];

    uint32_t w[16];
    memcpy(w, s, sizeof(w));

    for (int i = 0; i < 10; i++) {
        QR(w[0],w[4],w[ 8],w[12]); QR(w[1],w[5],w[ 9],w[13]);
        QR(w[2],w[6],w[10],w[14]); QR(w[3],w[7],w[11],w[15]);
        QR(w[0],w[5],w[10],w[15]); QR(w[1],w[6],w[11],w[12]);
        QR(w[2],w[7],w[ 8],w[13]); QR(w[3],w[4],w[ 9],w[14]);
    }
    for (int i = 0; i < 16; i++) out[i] = w[i] + s[i];
}

void chacha20_encrypt_ref(const uint32_t key[8], const uint32_t nonce[3],
                          uint32_t counter, const uint8_t *pt,
                          uint8_t *ct, int len) {
    uint32_t ks[16];
    int off = 0;
    uint32_t ctr = counter;
    while (off + 64 <= len) {
        chacha20_block(ks, key, ctr, nonce);
        const uint8_t *k = (const uint8_t *)ks;
        for (int i = 0; i < 64; i++) ct[off+i] = pt[off+i] ^ k[i];
        ctr++; off += 64;
    }
    if (off < len) {
        chacha20_block(ks, key, ctr, nonce);
        const uint8_t *k = (const uint8_t *)ks;
        for (int i = 0; off+i < len; i++) ct[off+i] = pt[off+i] ^ k[i];
    }
}
```

- [ ] **Step 4.1: Write chacha20_fused.ea**

Same rotation helpers and double-round logic as chacha20.ea. The encrypt loop adds statistics computation on each plaintext chunk:

```ea
// chacha20_fused.ea — Encrypt + plaintext statistics in one memory pass.

func rotl16(v: i32x4) -> i32x4 {
    return (v .<< splat(16)) .| (v .>> splat(16))
}
func rotl12(v: i32x4) -> i32x4 {
    return (v .<< splat(12)) .| (v .>> splat(20))
}
func rotl8(v: i32x4) -> i32x4 {
    return (v .<< splat(8)) .| (v .>> splat(24))
}
func rotl7(v: i32x4) -> i32x4 {
    return (v .<< splat(7)) .| (v .>> splat(25))
}

// Internal block function (same as chacha20.ea)
func block(key: *i32, nonce: *i32, counter: i32, out: *mut i32) {
    let c: i32x4 = [0x61707865, 0x3320646e, 0x79622d32, 0x6b206574]i32x4
    let k0: i32x4 = load(key, 0)
    let k1: i32x4 = load(key, 4)
    let cn: i32x4 = [counter, nonce[0], nonce[1], nonce[2]]i32x4

    store(out, 0, c)
    store(out, 4, k0)
    store(out, 8, k1)
    store(out, 12, cn)

    let mut i: i32 = 0
    while i < 10 {
        let mut r0: i32x4 = load(out, 0)
        let mut r1: i32x4 = load(out, 4)
        let mut r2: i32x4 = load(out, 8)
        let mut r3: i32x4 = load(out, 12)

        r0 = r0 .+ r1
        r3 = rotl16(r3 .^ r0)
        r2 = r2 .+ r3
        r1 = rotl12(r1 .^ r2)
        r0 = r0 .+ r1
        r3 = rotl8(r3 .^ r0)
        r2 = r2 .+ r3
        r1 = rotl7(r1 .^ r2)

        r1 = shuffle(r1, [1, 2, 3, 0])
        r2 = shuffle(r2, [2, 3, 0, 1])
        r3 = shuffle(r3, [3, 0, 1, 2])

        r0 = r0 .+ r1
        r3 = rotl16(r3 .^ r0)
        r2 = r2 .+ r3
        r1 = rotl12(r1 .^ r2)
        r0 = r0 .+ r1
        r3 = rotl8(r3 .^ r0)
        r2 = r2 .+ r3
        r1 = rotl7(r1 .^ r2)

        r1 = shuffle(r1, [3, 0, 1, 2])
        r2 = shuffle(r2, [2, 3, 0, 1])
        r3 = shuffle(r3, [1, 2, 3, 0])

        store(out, 0, r0)
        store(out, 4, r1)
        store(out, 8, r2)
        store(out, 12, r3)

        i = i + 1
    }

    store(out, 0, load(out, 0) .+ c)
    store(out, 4, load(out, 4) .+ k0)
    store(out, 8, load(out, 8) .+ k1)
    store(out, 12, load(out, 12) .+ cn)
}

export func chacha20_encrypt_stats(
    key: *i32, nonce: *i32, counter: i32,
    plaintext: *u8, ciphertext: *mut u8, len: i32,
    ks_i32: *mut i32, ks_u8: *u8,
    out_sum: *mut i64, out_count: *mut i32,
    out_min: *mut u8, out_max: *mut u8
) {
    let mut total_sum: i64 = 0
    let mut total_count: i32 = 0
    let mut gmin: u8 = 255
    let mut gmax: u8 = 0

    let mut offset: i32 = 0
    let mut ctr: i32 = counter

    while offset + 64 <= len {
        block(key, nonce, ctr, ks_i32)

        // Process 4 x 16-byte chunks: XOR + stats
        let pt0: u8x16 = load(plaintext, offset)
        let ks0: u8x16 = load(ks_u8, 0)
        store(ciphertext, offset, pt0 .^ ks0)

        let pt1: u8x16 = load(plaintext, offset + 16)
        let ks1: u8x16 = load(ks_u8, 16)
        store(ciphertext, offset + 16, pt1 .^ ks1)

        let pt2: u8x16 = load(plaintext, offset + 32)
        let ks2: u8x16 = load(ks_u8, 32)
        store(ciphertext, offset + 32, pt2 .^ ks2)

        let pt3: u8x16 = load(plaintext, offset + 48)
        let ks3: u8x16 = load(ks_u8, 48)
        store(ciphertext, offset + 48, pt3 .^ ks3)

        // Stats: min/max via reduce_min/reduce_max on each u8x16
        let min0: u8 = reduce_min(pt0)
        let min1: u8 = reduce_min(pt1)
        let min2: u8 = reduce_min(pt2)
        let min3: u8 = reduce_min(pt3)
        let mut bmin: u8 = min0
        if min1 < bmin { bmin = min1 }
        if min2 < bmin { bmin = min2 }
        if min3 < bmin { bmin = min3 }
        if bmin < gmin { gmin = bmin }

        let max0: u8 = reduce_max(pt0)
        let max1: u8 = reduce_max(pt1)
        let max2: u8 = reduce_max(pt2)
        let max3: u8 = reduce_max(pt3)
        let mut bmax: u8 = max0
        if max1 > bmax { bmax = max1 }
        if max2 > bmax { bmax = max2 }
        if max3 > bmax { bmax = max3 }
        if bmax > gmax { gmax = bmax }

        // Stats: sum — widen u8 to i32, reduce, accumulate to i64
        // widen_u8_i32x4 takes first 4 bytes of u8x16
        // Need 4 calls + shuffles per u8x16 to sum all 16 bytes
        // Simplified: use reduce_add which returns u8, then widen to i64
        // Actually reduce_add on u8x16 wraps at 255. Use widen approach.
        // For now, accumulate per-byte in scalar from reduce values
        // TODO: implementer should optimize with proper widen+reduce chain
        let s0: i32 = to_i32(reduce_add(widen_u8_i32x4(pt0)))
        let s1: i32 = to_i32(reduce_add(widen_u8_i32x4(pt1)))
        let s2: i32 = to_i32(reduce_add(widen_u8_i32x4(pt2)))
        let s3: i32 = to_i32(reduce_add(widen_u8_i32x4(pt3)))
        // NOTE: widen_u8_i32x4 only sums bytes 0-3 of each u8x16!
        // The implementer MUST fix this to sum all 16 bytes per vector.
        // Approach: shuffle to bring bytes 4-7, 8-11, 12-15 into position,
        // call widen_u8_i32x4 on each, reduce_add, accumulate.
        total_sum = total_sum + to_i64(s0) + to_i64(s1) + to_i64(s2) + to_i64(s3)

        total_count = total_count + 64
        ctr = ctr + 1
        offset = offset + 64
    }

    // Scalar tail — stats only, encryption handled by masked SIMD
    if offset < len {
        block(key, nonce, ctr, ks_i32)
        let remaining: i32 = len - offset

        let mut tail_off: i32 = 0
        while tail_off + 16 <= remaining {
            let pt: u8x16 = load(plaintext, offset + tail_off)
            let ks: u8x16 = load(ks_u8, tail_off)
            store(ciphertext, offset + tail_off, pt .^ ks)

            let tmin: u8 = reduce_min(pt)
            let tmax: u8 = reduce_max(pt)
            if tmin < gmin { gmin = tmin }
            if tmax > gmax { gmax = tmax }
            // Same widen issue for sum — implementer must fix
            total_sum = total_sum + to_i64(to_i32(reduce_add(widen_u8_i32x4(pt))))
            total_count = total_count + 16

            tail_off = tail_off + 16
        }

        if tail_off < remaining {
            let left: i32 = remaining - tail_off
            let pt: u8x16 = load_masked(plaintext, offset + tail_off, left)
            let ks: u8x16 = load_masked(ks_u8, tail_off, left)
            store_masked(ciphertext, offset + tail_off, pt .^ ks, left)

            let tmin: u8 = reduce_min(pt)
            let tmax: u8 = reduce_max(pt)
            if tmin < gmin { gmin = tmin }
            if tmax > gmax { gmax = tmax }
            total_sum = total_sum + to_i64(to_i32(reduce_add(widen_u8_i32x4(pt))))
            total_count = total_count + left
        }
    }

    out_sum[0] = total_sum
    out_count[0] = total_count
    out_min[0] = gmin
    out_max[0] = gmax
}
```

```ea
let mid: i32x4 = widen_u8_i32x4(shifted)
// ... etc for bytes 8-11 and 12-15
```

```python
#!/usr/bin/env python3
"""ChaCha20 benchmark — honest comparison."""

import ctypes as ct
import numpy as np
import struct
import time
import os
import platform
import sys
from pathlib import Path
from concurrent.futures import ThreadPoolExecutor

DATA_SIZE = 64 * 1024 * 1024  # 64 MB
WARMUP = 3
RUNS = 10
BLOCK = 64

def to_i32_array(values):
    arr = (ct.c_int32 * len(values))()
    for i, v in enumerate(values):
        arr[i] = ct.c_int32(v & 0xFFFFFFFF).value
    return arr

def median(vals):
    s = sorted(vals)
    n = len(s)
    return s[n // 2] if n % 2 else (s[n//2-1] + s[n//2]) / 2

def bench(name, func):
    for _ in range(WARMUP):
        func()
    times = []
    for _ in range(RUNS):
        t0 = time.perf_counter()
        func()
        times.append(time.perf_counter() - t0)
    gbps = [DATA_SIZE / t / 1e9 for t in times]
    med = median(gbps)
    std = float(np.std(gbps))
    print(f"  {name:45s}  {med:8.2f} GB/s  (std: {std:.2f})")
    return med, std

# Load libs
ea = ct.CDLL(str(Path(__file__).parent / "libchacha20.so"))
ref = ct.CDLL(str(Path(__file__).parent / "libchacha20_ref.so"))

ea.chacha20_encrypt.argtypes = [
    ct.POINTER(ct.c_int32), ct.POINTER(ct.c_int32), ct.c_int32,
    ct.POINTER(ct.c_uint8), ct.POINTER(ct.c_uint8), ct.c_int32,
    ct.POINTER(ct.c_int32), ct.POINTER(ct.c_uint8),
]
ea.chacha20_encrypt.restype = None

ref.chacha20_encrypt_ref.argtypes = [

    ct.POINTER(ct.c_uint8), ct.POINTER(ct.c_uint8), ct.c_int32,
]
ref.chacha20_encrypt_ref.restype = None

# Data
np.random.seed(42)
plaintext = np.random.randint(0, 256, size=DATA_SIZE, dtype=np.uint8)
ciphertext = np.zeros(DATA_SIZE, dtype=np.uint8)

key_bytes = bytes(range(32))
nonce_bytes = bytes(12)
key_arr = to_i32_array(list(struct.unpack("<8I", key_bytes)))
nonce_arr = to_i32_array(list(struct.unpack("<3I", nonce_bytes)))

pt_ptr = plaintext.ctypes.data_as(ct.POINTER(ct.c_uint8))
ct_ptr = ciphertext.ctypes.data_as(ct.POINTER(ct.c_uint8))

# Scratch buffer (shared i32/u8 views)
scratch = (ct.c_uint8 * 64)()
ks_i32 = ct.cast(scratch, ct.POINTER(ct.c_int32))
ks_u8 = ct.cast(scratch, ct.POINTER(ct.c_uint8))

print("=" * 70)

print(f"Cores:     {os.cpu_count()}")
print(f"Data:      {DATA_SIZE // 1024 // 1024} MB")
print(f"Runs:      {RUNS} (median)")
print("=" * 70)

# NumPy XOR (baseline)
key_rep = np.tile(np.frombuffer(key_bytes, dtype=np.uint8), DATA_SIZE // 32 + 1)[:DATA_SIZE]
bench("NumPy XOR (not real crypto)", lambda: np.bitwise_xor(plaintext, key_rep, out=ciphertext))

# Generic C
bench("Generic C (-O3, no SIMD)",
      lambda: ref.chacha20_encrypt_ref(key_arr, nonce_arr, 0, pt_ptr, ct_ptr, DATA_SIZE))

try:

    nonce_ssl = b'\x00\x00\x00\x00' + nonce_bytes
    pt_bytes = bytes(plaintext)
    def run_openssl():

except ImportError:

bench("Ea ChaCha20 (single core)",

# Ea parallel
num_threads = os.cpu_count()
blocks_total = DATA_SIZE // BLOCK

def ea_parallel():
    blocks_per = blocks_total // num_threads
    rem = blocks_total % num_threads
    def work(tid):
        start = tid * blocks_per + min(tid, rem)
        n = blocks_per + (1 if tid < rem else 0)
        boff = start * BLOCK
        blen = n * BLOCK
        if blen == 0: return
        # Each thread needs its own scratch buffer
        tsc = (ct.c_uint8 * 64)()
        ti32 = ct.cast(tsc, ct.POINTER(ct.c_int32))
        tu8 = ct.cast(tsc, ct.POINTER(ct.c_uint8))
        src = ct.cast(ct.addressof(ct.c_uint8.from_buffer(plaintext, boff)), ct.POINTER(ct.c_uint8))
        dst = ct.cast(ct.addressof(ct.c_uint8.from_buffer(ciphertext, boff)), ct.POINTER(ct.c_uint8))
        ea.chacha20_encrypt(key_arr, nonce_arr, start, src, dst, blen, ti32, tu8)
    with ThreadPoolExecutor(max_workers=num_threads) as pool:
        list(pool.map(work, range(num_threads)))

bench(f"Ea ChaCha20 parallel ({num_threads} cores)", ea_parallel)

    ocnt = (ct.c_int32 * 1)()
    omin = (ct.c_uint8 * 1)()

                                                ks_i32, ks_u8, osum, ocnt, omin, omax))
```
